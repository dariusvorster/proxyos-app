# V1.1 Self-Signed Upstream Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add three-way upstream protocol picker (`http` / `https-trusted` / `https-insecure`) to the expose wizard plus a server-side probe endpoint, so users can correctly proxy HTTPS upstreams with self-signed certificates.

**Architecture:** Add `upstream_protocol` and `upstream_sni` columns to the `routes` table. Update the Caddy transport block builder to use the enum instead of port-detection heuristics. Add a `probeUpstream` helper, two new tRPC endpoints (`probe`, `diagnostics`), a 3-way radio in the wizard, and a diagnostics panel on the route detail page.

**Tech Stack:** TypeScript, Drizzle ORM, SQLite (better-sqlite3), tRPC v11, Zod, Caddy JSON API, Node `tls` module, vitest, Next.js 15 App Router

```typescript
import * as tls from 'node:tls'

export type UpstreamProtocol = 'http' | 'https-trusted' | 'https-insecure'

export interface ProbeDetails {
  certCn?: string
  certIssuer?: string
  certExpiresAt?: string
  statusCode?: number
}

export interface ProbeResult {
  suggestion: UpstreamProtocol | null
  details?: ProbeDetails
  error?: string
}

function tlsConnect(host: string, port: number, rejectUnauthorized: boolean): Promise<tls.TLSSocket> {
  return new Promise((resolve, reject) => {
    const socket = tls.connect({ host, port, rejectUnauthorized, timeout: 3000 })
    socket.on('secureConnect', () => resolve(socket))
    socket.on('error', (err) => {
      socket.destroy()
      reject(err)
    })
    socket.on('timeout', () => {
      socket.destroy()
      reject(new Error('timeout'))
    })
  })
}

export async function probeUpstream(host: string, port: number): Promise<ProbeResult> {
  // 1. HTTPS strict verify
  try {
    const socket = await tlsConnect(host, port, true)
    const cert = socket.getPeerCertificate()
    socket.destroy()
    return {
      suggestion: 'https-trusted',
      details: {
        certCn: cert.subject?.CN,
        certIssuer: cert.issuer?.CN,
        certExpiresAt: cert.valid_to,
      },
    }
  } catch { /* fall through */ }

  // 2. HTTPS skip-verify
  try {
    const socket = await tlsConnect(host, port, false)
    const cert = socket.getPeerCertificate()
    socket.destroy()
    return {
      suggestion: 'https-insecure',
      details: {
        certCn: cert.subject?.CN,
        certIssuer: cert.issuer?.CN,
        certExpiresAt: cert.valid_to,
      },
    }
  } catch { /* fall through */ }

  // 3. Plain HTTP HEAD
  try {
    const res = await fetch(`http://${host}:${port}/`, {
      method: 'HEAD',
      signal: AbortSignal.timeout(3000),
    })
    return { suggestion: 'http', details: { statusCode: res.status } }
  } catch { /* fall through */ }

  return { suggestion: null, error: 'Could not connect to upstream.' }
}
```

```typescript
type UpstreamProtocol = 'http' | 'https-trusted' | 'https-insecure'

function buildTransport(
  upstreamProtocol: UpstreamProtocol,
  upstreamSni?: string | null,
): Record<string, unknown> | undefined {
  switch (upstreamProtocol) {
    case 'http':
      return undefined
    case 'https-trusted':
      return {
        protocol: 'http',
        tls: upstreamSni ? { server_name: upstreamSni } : {},
      }
    case 'https-insecure':
      return {
        protocol: 'http',
        tls: { insecure_skip_verify: true },
      }
  }
}
```
